/* ============================================================
   Nerta Brasil — Typographic System
   Source: PDF §3 Sistema Tipográfico

   ============================================================ */

  /* ---- Families ---- */

  --size-h2: 28px;
  --size-h3: 22px;
  --size-h4: 18px;

  --size-sm: 13px;
  --size-xs: 11px;
  --size-label: 10px;     /* uppercase section labels */

  /* ---- Line heights ---- */
  --leading-tight: 1.12;  /* headlines */
  --leading-snug: 1.3;

  --type-label-tracking: var(--tracking-label);
